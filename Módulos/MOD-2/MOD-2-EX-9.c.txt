#include<msp430.h>

#define TRUE    1
#define FALSE   0
#define MAX     32767
#define MEIOSEG 16

void io_config ();

void io_config (void) {
    P1DIR |= BIT0;
    P1OUT &= ~BIT0;
}

void timer_config (void) {
    TA0CTL = TASSEL_2   |   // Sel. SMCLK
             ID_0       |   // Div por 1
             MC_1       |   // Modo 1
             TACLR;         // Come√ßando Zerado
    TA0EX0 = 0;             // Dividir por 1 (TAIDEX = 0)
    TA0CCR0 = MAX;
}

int main (void){
    WDTCTL = WDTPW | WDTHOLD;
    unsigned int cont = 0;

    io_config();
    timer_config ();

    while (TRUE){
        while((TA0CTL & TAIFG)== 0); //Esperar TAIFG
        TA0CTL &= ~TAIFG;
        cont++;
        if (cont == MEIOSEG){
            cont = 0;
            P1OUT^=BIT0;
        }
    }
    return 0;
}
